## Install

```bash
npm install --save fluree-admin-components
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'fluree-admin-components'
import 'fluree-admin-components/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
